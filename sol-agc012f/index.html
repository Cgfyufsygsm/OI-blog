using fastIO::is;
using fastIO::os;
using mint = atcoder::modint1000000007;
const int maxn = 105;
int n, a[maxn];
mint f[maxn][maxn][maxn];

int main() {
    is >> n;
    int m = 2 * n - 1;
    FOR(i, 1, m) is >> a[i];
    sort(a + 1, a + m + 1);
    f[n][0][0] = 1;
    DEC(i, n, 2) {
        const int &l = (a[i] != a[i - 1]), &r = (a[m - i + 1] != a[m - i + 2]);

            FOR(k, 0, m) {

                FOR(t, 0, k + r - 1) f[i - 1][j + l + 1][t] += f[i][j][k];
            }
        }
    }
    mint ans = 0;
    FOR(i, 0, m) FOR(j, 0, m) ans += f[1][i][j];
