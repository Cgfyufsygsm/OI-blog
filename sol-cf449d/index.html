const int maxn = 2e6 + 5;
modint f[maxn], pow[maxn];
int n;

void AND(modint *f, int lim, modint x = 1) {
    for (int o = 2, k = 1; o <= lim; o <<= 1, k <<= 1)
        for (int i = 0; i < lim; i += o)
            for (int j = 0; j < k; ++j)
                f[i + j] += f[i + j + k] * x;
    return;
}

    FOR(i, 1, n) {
