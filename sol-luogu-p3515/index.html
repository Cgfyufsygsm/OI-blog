typedef double db;

int a[maxn], n;
db p[maxn], sq[maxn];

void solve(int l, int r, int L, int R) {
    if (l > r) return;
    int mid = (l + r) >> 1, k;
    db mx = 0, tmp;
    for (int j = L; j <= mid && j <= R; ++j)
        if ((tmp = a[j] + sq[mid - j]) > mx) mx = tmp, k = j;

    solve(l, mid - 1, L, k);
    solve(mid + 1, r, k, R);
    return;
}

int bound(int x, int y) {
    int l = y, r = k[x] ? k[x] : n, mid, ret = r + 1;
    while (l <= r) {
        mid = (l + r) >> 1;
        if (calc(mid, x) <= calc(mid, y))

void solve() {
