    FOR(i, 1, n) {
        FOR(j, 1, i - 1) {
            char ch; cin >> ch;
            if (ch == '1') a[i][j] = 1;
            a[j][i] = a[i][j] ^ 1;

    FOR(i, 1, n) L[i][i] = R[i][i] = 1;
