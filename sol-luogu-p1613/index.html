typedef long long ll;

inline ll read()
{
    char c = getchar();
    ll s = 0;
    while (!isdigit(c))
        c = getchar();
    while (isdigit(c))
        s = 10 * s + c - '0', c = getchar();
    return s;
}

inline ll min(ll a,ll b)
{
    return a<b?a:b;
}

const int maxn=53;
ll dis[maxn][maxn],n,m;

{

        int x=read(),y=read();

        dis[x][y]=1;

        for(int x=1;x<=n;x++)
            for(int y=1;y<=n;y++)
