#include <cstdio>
#include <cstring>
#define il inline
#define FOR(i, a, b) for (int i = a; i <= b; ++i)
#define DEC(i, a, b) for (int i = a; i >= b; --i)

typedef long double db;
const int maxn = 1e5 + 5;
const db EPS = 1e-9;

int n;
db a[maxn], b[maxn], r[maxn];
db f[maxn], K[maxn], B[maxn];

int t[maxn], ls[maxn], rs[maxn], cnt, root;

il db calc(int i, db X) {return K[i] * X + B[i];}
il db max(db a, db b) {return a > b ? a : b;}

#define M ((i + j) / 2.0)
#define L ls[k]
#define R rs[k]

void insert(db i, db j, int &k, int u) {
    if (j - i <= EPS) return;
    if (!k) k = ++cnt;
    int v = t[k];
    db resu = calc(u, M), resv = calc(v, M);
    if (K[u] > K[v]) {
        if (resu > resv)
            insert(i, M, L, v), t[k] = u;
        else insert(M + 1, j, R, u);
    }
    else {
        if (resu > resv)
            insert(M + 1, j, R, v), t[k] = u;
        else insert(i, M, L, u);
    }
    return;
}

db query(db i, db j, int k, db X) {
    db ret = calc(t[k], X);
    if (j - i < EPS) return ret;
    if (X <= M) return max(ret, query(i, M, L, X));
    else return max(ret, query(M + 1, j, R, X));
}

int main() {
    scanf("%d %Lf", &n, f + 1);
    FOR(i, 1, n) scanf("%Lf %Lf %Lf", a + i, b + i, r + i);
    K[1] = (f[1] * r[1]) / (r[1] * a[1] + b[1]), B[1] = f[1] / (r[1] * a[1] + b[1]);
    insert(0, 10, root, 1);
    FOR(i, 2, n) {
        f[i] = max(f[i - 1], b[i] * query(0, 10, root, a[i] / b[i]));
        K[i] = f[i] * r[i] / (r[i] * a[i] + b[i]), B[i] = f[i] / (r[i] * a[i] + b[i]);
        insert(0, 10, root, i);
        //printf("i %d %.3Lf\n", i, f[i]);
    }
    printf("%.3Lf\n", f[n]);
    return 0;
}

typedef double db;
const int maxn = 1e5 + 5;

int n;
db a[maxn], b[maxn], r[maxn], c[maxn], d[maxn];
db f[maxn], K[maxn], B[maxn];

int t[maxn << 2];

il db calc(int i, int u) {return K[i] * c[u] + B[i];}
il db max(db a, db b) {return a > b ? a : b;}

#define M ((i + j) >> 1)
#define L (k << 1)
#define R (L | 1)

void insert(int i, int j, int k, int u) {
    int v = t[k];
    db resu = calc(u, M), resv = calc(v, M);
    if (i == j) {
        if (resu > resv) t[k] = u;
        return;
    }
    if (K[u] > K[v]) {
        if (resu > resv)
            insert(i, M, L, v), t[k] = u;
        else insert(M + 1, j, R, u);
    }
    else {
        if (resu > resv)
            insert(M + 1, j, R, v), t[k] = u;
        else insert(i, M, L, u);
    }
    return;
}

db query(int i, int j, int k, int u) {
    db ret = calc(t[k], u);
    if (i == j) return ret;
    if (u <= M) return max(ret, query(i, M, L, u));
    else return max(ret, query(M + 1, j, R, u));
}
